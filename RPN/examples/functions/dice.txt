d4   = 4 dieroll
d6   = 6 dieroll
d8   = 8 dieroll
d10  = 10 dieroll
d20  = 20 dieroll
d100 = 100 dieroll

keep2 = pop
keeptop = exch pop
max2 = over over <= "keeptop" "keep2" ifelse
min2 = over over >= "keeptop" "keep2" ifelse
dis = d20 d20 min2
adv = d20 d20 max2
