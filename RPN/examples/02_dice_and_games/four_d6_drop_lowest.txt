#------------------------------------------------------------
# Example: Four d6, Drop the Lowest
#
# Category:
#     02 Dice And Games
#
# Purpose:
#     Rolls four six-sided dice, drops the lowest roll, and adds the
#     remaining three to produce one role-playing ability score.
#
# Demonstrates:
#     dieroll, repeat, sort, pop, sum, sprintf, executable values,
#     user functions, named scratch stacks, withstack, withcopystack
#
# Notes:
#     The public functions use private named stacks so they neither consume
#     nor include values already on the current stack.
#
# Status:
#     Complete
#------------------------------------------------------------

# Roll one six-sided die.
d6 = 6 dieroll

# repeat consumes the count 4 and executes the quoted function name "d6"
# four times. The four results remain on the scratch stack.
roll_four_d6 = 4 "d6" repeat

# Roll and sort four dice. With ascending sort order, the lowest die is on
# top and the highest die is deepest.
ability_score_rolls = roll_four_d6 sort

# sort puts the lowest die on top. pop discards that die, and sum replaces
# the remaining three dice with their total.
drop_lowest_and_sum = pop sum

# Run drop_lowest_and_sum on a copy of the four sorted dice. withcopystack
# returns the total while preserving the original dice for later formatting.
ability_score_total = "_ability_total" "drop_lowest_and_sum" withcopystack

# This is the complete calculation, intended to run on an empty scratch
# stack. Its stack effect is:
#
#     ( empty -- score )
ability_score_work = ability_score_rolls drop_lowest_and_sum

# withstack switches to the named stack, runs ability_score_work there,
# returns its top result, and restores the caller's original stack. This
# keeps unrelated caller values out of sort and sum.
ability_score = "_ability_score" "ability_score_work" withstack

# Keep both the sorted rolls and their total:
#
#     ( empty -- lowest middle middle highest total )
ability_score_and_rolls = ability_score_rolls ability_score_total

# Function bodies are token strings, so a literal space cannot live inside
# one token. Build the format on a private stack using chr(32) for each space,
# then join the fragments with an empty delimiter.
ability_score_format_work = "%d" 32 chr "+" 32 chr "%d" 32 chr "+" 32 chr "%d" 32 chr "+" 32 chr "[%d]" 32 chr "=" 32 chr "%2d" "" join
ability_score_format = "_ability_score_format" "ability_score_format_work" withstack

# sprintf consumes the four dice, the total, and the constructed format.
# Because the dice are stored lowest-first on the stack, sprintf receives the
# three kept dice in descending order and the dropped die last.
ability_score_as_string_work = ability_score_and_rolls ability_score_format sprintf

# Run the formatter on its own scratch stack and return only the string.
ability_score_as_string = "_ability_score_string" "ability_score_as_string_work" withstack

# Example:
#
#     ability_score
#
# Possible result:
#
#     15
#
#     ability_score_as_string
#
# Possible result:
#
#     6 + 5 + 4 + [2] = 15
#------------------------------------------------------------
